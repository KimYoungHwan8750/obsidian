```
const { BrowserWindow } = require('electron');

let win = new BrowserWindow({
  width: 400,
  height: 100,
  transparent: true,    // 배경 투명화
  frame: false,          // 타이틀바 제거
  alwaysOnTop: true,     // 다른 창 위에 항상 표시
  skipTaskbar: true,     // 작업 표시줄에 아이콘 숨기기
  focusable: false,      // 클릭이나 포커스를 뺏지 않음 (중요!)
  webPreferences: {
    nodeIntegration: true
  }
});

// HTML/CSS에서 opacity를 주어 반투명하게 디자인
win.loadURL('...');
```
