```js
$DOM.classList
```
DomTokenList을 반환한다.


#### 함수

```js
//add 클래스를 추가한다
$DOM.classList.add("className1");
$DOM.classList.add("className1","className2");

//remove 클래스를 삭제한다
$DOM.classList.remove("className1");
$DOM.classList.remove("className1","className2");

//replace 클래스를 교체한다
$DOM.classList.replace("className1","className3");

//toggle 클래스가 있으면 삭제하고 없으면 생성한다
$DOM.classList.toggle("className1");

//contains 해당 클래스가 있는지 확인한다
$DOM.classList.contains("className1");



```
